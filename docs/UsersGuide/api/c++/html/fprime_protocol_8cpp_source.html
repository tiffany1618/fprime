24 void FprimeFraming::frame(const U8* const data, const U32 size, Fw::ComPacket::ComPacketType packet_type) {
25  FW_ASSERT(data != nullptr);
26  FW_ASSERT(m_interface != nullptr);
27  // Use of I32 size is explicit as ComPacketType will be specifically serialized as an I32
28  FpFrameHeader::TokenType real_data_size = size + ((packet_type != Fw::ComPacket::FW_PACKET_UNKNOWN) ? sizeof(I32) : 0);
29  FpFrameHeader::TokenType total = real_data_size + FpFrameHeader::SIZE + HASH_DIGEST_LENGTH;
30  Fw::Buffer buffer = m_interface->allocate(total);
31  Fw::SerializeBufferBase& serializer = buffer.getSerializeRepr();
32  Utils::HashBuffer hash;
33 
34  // Serialize data
35  Fw::SerializeStatus status;
36  status = serializer.serialize(FpFrameHeader::START_WORD);
37  FW_ASSERT(status == Fw::FW_SERIALIZE_OK, status);
38 
39  status = serializer.serialize(real_data_size);
40  FW_ASSERT(status == Fw::FW_SERIALIZE_OK, status);
41 
42  // Serialize packet type if supplied, otherwise it *must* be present in the data
43  if (packet_type != Fw::ComPacket::FW_PACKET_UNKNOWN) {
44  status = serializer.serialize(static_cast<I32>(packet_type)); // I32 used for enum storage
45  FW_ASSERT(status == Fw::FW_SERIALIZE_OK, status);
46  }
47 
48  status = serializer.serialize(data, size, true); // Serialize without length
49  FW_ASSERT(status == Fw::FW_SERIALIZE_OK, status);
50 
51  // Calculate and add transmission hash
52  Utils::Hash::hash(buffer.getData(), total - HASH_DIGEST_LENGTH, hash);
53  status = serializer.serialize(hash.getBuffAddr(), HASH_DIGEST_LENGTH, true);
54  FW_ASSERT(status == Fw::FW_SERIALIZE_OK, status);
55 
56  buffer.setSize(total);
57 
58  m_interface->send(buffer);
59 }

61 bool FprimeDeframing::validate(Types::CircularBuffer& ring, U32 size) {
62  Utils::Hash hash;
63  Utils::HashBuffer hashBuffer;
64  // Initialize the checksum and loop through all bytes calculating it
65  hash.init();
66  for (U32 i = 0; i < size; i++) {
67  char byte;
68  const Fw::SerializeStatus status = ring.peek(byte, i);
69  FW_ASSERT(status == Fw::FW_SERIALIZE_OK, status);
70  hash.update(&byte, 1);
71  }
72  hash.final(hashBuffer);
73  // Now loop through the hash digest bytes and check for equality
74  for (U32 i = 0; i < HASH_DIGEST_LENGTH; i++) {
75  char calc = static_cast<char>(hashBuffer.getBuffAddr()[i]);
76  char sent = 0;
77  const Fw::SerializeStatus status = ring.peek(sent, size + i);
78  FW_ASSERT(status == Fw::FW_SERIALIZE_OK, status);
79  if (calc != sent) {
80  return false;
81  }
82  }
83  return true;
84 }

86 DeframingProtocol::DeframingStatus FprimeDeframing::deframe(Types::CircularBuffer& ring, U32& needed) {
87  FpFrameHeader::TokenType start = 0;
88  FpFrameHeader::TokenType size = 0;
89  FW_ASSERT(m_interface != nullptr);
90  // Check for header or ask for more data
91  if (ring.get_allocated_size() < FpFrameHeader::SIZE) {
92  needed = FpFrameHeader::SIZE;
93  return DeframingProtocol::DEFRAMING_MORE_NEEDED;
94  }
95  // Read start value from header
96  Fw::SerializeStatus status = ring.peek(start, 0);
97  FW_ASSERT(status == Fw::FW_SERIALIZE_OK, status);
98  if (start != FpFrameHeader::START_WORD) {
99  // Start word must be valid
100  return DeframingProtocol::DEFRAMING_INVALID_FORMAT;
101  }
102  // Read size from header
103  status = ring.peek(size, sizeof(FpFrameHeader::TokenType));
104  FW_ASSERT(status == Fw::FW_SERIALIZE_OK, status);
105  const U32 maxU32 = std::numeric_limits<U32>::max();
106  if (size > maxU32 - (FpFrameHeader::SIZE + HASH_DIGEST_LENGTH)) {
107  // Size is too large to process: needed would overflow
108  return DeframingProtocol::DEFRAMING_INVALID_SIZE;
109  }
110  needed = (FpFrameHeader::SIZE + size + HASH_DIGEST_LENGTH);
111  // Check frame size
112  const U32 frameSize = size + FpFrameHeader::SIZE + HASH_DIGEST_LENGTH;
113  if (frameSize > ring.get_capacity()) {
114  // Frame size is too large for ring buffer
115  return DeframingProtocol::DEFRAMING_INVALID_SIZE;
116  }
117  // Check for enough data to deserialize everything;
118  // otherwise break and wait for more.
119  else if (ring.get_allocated_size() < needed) {
120  return DeframingProtocol::DEFRAMING_MORE_NEEDED;
121  }
122  // Check the checksum
123  if (not this->validate(ring, needed - HASH_DIGEST_LENGTH)) {
124  return DeframingProtocol::DEFRAMING_INVALID_CHECKSUM;
125  }
126  Fw::Buffer buffer = m_interface->allocate(size);
127  // Some allocators may return buffers larger than requested.
128  // That causes issues in routing; adjust size.
129  FW_ASSERT(buffer.getSize() >= size);
130  buffer.setSize(size);
131  status = ring.peek(buffer.getData(), size, FpFrameHeader::SIZE);
132  FW_ASSERT(status == Fw::FW_SERIALIZE_OK, status);
133  m_interface->route(buffer);
134  return DeframingProtocol::DEFRAMING_STATUS_SUCCESS;
135 }
